\[
\varphi = \frac{W}{q},
\]

\subsubsection*{Связь напряжённости и потенциала}

\[
E = -\frac{\Delta \varphi}{\Delta x},
\quad
\Delta\varphi = Ed,
\quad
E = k \frac{Q}{r^2},
\quad
\varphi = k \frac{Q}{r}.
\]
