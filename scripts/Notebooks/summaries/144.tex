\[
  \Delta L = \alpha L_0 \Delta T, \quad L = L_0 + \Delta L,
\]

  \Delta V = \beta V_0 \Delta T, \quad V = V_0 + \Delta V.
\]

\subsubsection*{Связь коэффициентов}

\[
  \beta \approx 3\alpha.
