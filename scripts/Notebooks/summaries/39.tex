\[
m\vec a = \vec F, \quad \vec a = \frac{d\vec v}{dt}
\]

\[
\vec a = \vec a_0 + \vec a', \quad m\vec a' = \vec F - m\vec a_0
\]

\[
\vec F_{\rm ин} = -m\vec a_0, \quad m\vec a' = \vec F + \vec F_{\rm ин}
\]

\[
