\]

\[

\]

\subsubsection*{Столкновение под углом с движущейся стенкой}
\[
\vec v = \vec v_\tau + 2\vec V - \vec v_{0n},
\quad
\tan\theta_{\rm после} = \frac{|\vec v_\tau|}{|2\vec V - \vec v_{0n}|} = \frac{|\vec v_{0n}|}{|2\vec V - \vec v_{0n}|}\tan\theta_{\rm до}.
\]
