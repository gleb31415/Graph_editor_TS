\subsubsection*{Заряд}

Элементарный заряд:
\[
e = 1.6\cdot10^{-19}\ \mathrm{Кл}
\]
Дискретность заряда:
\[
Q = ne,
\quad n \in \mathbb{Z}
\]

\subsubsection*{Закон Кулона}

\[
F = k \frac{Q_{1} Q_{2}}{r^{2}},
\quad k \approx 9\cdot10^{9}\ \mathrm{Н\cdot м^{2}/Кл^{2}}
\]

\subsubsection*{Потенциальная энергия взаимодействия зарядов}

\[
U(r) = k \frac{Q_1 Q_2}{r}
\]
