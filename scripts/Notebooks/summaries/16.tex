\subsubsection*{Левая и правая ветви}

\[
T_1R - T_2R = 0
\quad\Longrightarrow\quad
T_1 = T_2 = T.
\]

\subsubsection*{Центральная и боковые нити}

\[
T_0 - 2T = 0
\quad\Longrightarrow\quad
T_0 = 2T.
\]

\textbf{Пример.} 

\[
T_0 = mg,
\quad
T = \frac{mg}{2}.
\]
