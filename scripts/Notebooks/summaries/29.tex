\subsubsection*{Теорема Кёнига для кинетической энергии}

\[
\vec V = \frac{1}{M}\sum_i m_i\vec v_i, \quad M = \sum_i m_i,
\quad
\vec v_i = \vec V + \delta\vec v_i,
\]

\[
T = \frac12 M |\vec V|^2 + \sum_i \frac12 m_i |\delta \vec v_i|^2.
\]
